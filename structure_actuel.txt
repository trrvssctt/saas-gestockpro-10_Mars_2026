ALTER TABLE tenants 
  ADD COLUMN IF NOT EXISTS last_payment_amount NUMERIC(15, 2) DEFAULT 0,
  ADD COLUMN IF NOT EXISTS pending_plan_id VARCHAR(50),
  ADD COLUMN IF NOT EXISTS pending_period VARCHAR(10);


CREATE TABLE IF NOT EXISTS announcements (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title VARCHAR(255) NOT NULL,
  body TEXT NOT NULL,
  type VARCHAR(20) DEFAULT 'INFO' CHECK (type IN ('INFO','WARNING','UPDATE','PROMO','MAINTENANCE')),
  target_plan VARCHAR(50),
  is_active BOOLEAN DEFAULT true,
  is_pinned BOOLEAN DEFAULT false,
  expires_at TIMESTAMP WITH TIME ZONE,
  created_by UUID REFERENCES users(id) ON DELETE SET NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);


CREATE TABLE IF NOT EXISTS hr_rules (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id UUID NOT NULL,
  name VARCHAR(200) NOT NULL,
  description TEXT,
  type VARCHAR(20) NOT NULL,
  condition_operator VARCHAR(5) DEFAULT 'GT',
  condition_value NUMERIC(10,2) DEFAULT 0,
  condition_unit VARCHAR(10) DEFAULT 'MINUTES',
  action_type VARCHAR(30) NOT NULL,
  action_value NUMERIC(10,2) DEFAULT 0,
  is_active BOOLEAN DEFAULT true,
  sort_order INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
